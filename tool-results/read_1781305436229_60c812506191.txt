     1→'use client';
     2→
     3→import React, { useState, useEffect, useCallback } from 'react';
     4→import {
     5→  ArrowLeft, Edit2, Save, X, FileText, Briefcase, Heart, DollarSign,
     6→  Palmtree, FolderOpen, Loader2, AlertCircle, Plus, Clock
     7→} from 'lucide-react';
     8→import { Button } from '@/components/ui/button';
     9→import { Input } from '@/components/ui/input';
    10→import { Label } from '@/components/ui/label';
    11→import { Badge } from '@/components/ui/badge';
    12→import { Card, CardContent, CardHeader, CardTitle, CardDescription } from '@/components/ui/card';
    13→import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
    14→import {
    15→  Table, TableBody, TableCell, TableHead, TableHeader, TableRow,
    16→} from '@/components/ui/table';
    17→import {
    18→  Select, SelectContent, SelectItem, SelectTrigger, SelectValue,
    19→} from '@/components/ui/select';
    20→import {
    21→  Dialog, DialogContent, DialogHeader, DialogTitle, DialogDescription,
    22→} from '@/components/ui/dialog';
    23→import { Separator } from '@/components/ui/separator';
    24→import { Skeleton } from '@/components/ui/skeleton';
    25→import { useToast } from '@/hooks/use-toast';
    26→
    27→type UserRole = 'ADMIN' | 'ANALISTA' | 'APROBADOR' | 'GERENCIA' | 'AUDITOR' | 'EMPLEADO';
    28→
    29→interface EmployeeDetailProps {
    30→  empleadoId: string;
    31→  onBack: () => void;
    32→  userRole: UserRole;
    33→  accessToken: string | null;
    34→}
    35→
    36→interface EmpleadoDetail {
    37→  id: string;
    38→  codigo_empleado: string;
    39→  primer_nombre: string;
    40→  segundo_nombre: string | null;
    41→  primer_apellido: string;
    42→  segundo_apellido: string | null;
    43→  apellido_casada: string | null;
    44→  dui: string;
    45→  nit: string | null;
    46→  fecha_nacimiento: string | null;
    47→  genero: string | null;
    48→  estado_civil: string | null;
    49→  direccion: string | null;
    50→  telefono: string | null;
    51→  email_personal: string | null;
    52→  numero_isss: string | null;
    53→  numero_afp: string | null;
    54→  afp_administradora: string | null;
    55→  tipo_sangre: string | null;
    56→  contacto_emergencia_nombre: string | null;
    57→  contacto_emergencia_telefono: string | null;
    58→  contacto_emergencia_relacion: string | null;
    59→  nacionalidad: string;
    60→  fecha_ingreso: string;
    61→  fecha_salida: string | null;
    62→  salario_base: number;
    63→  estado: string;
    64→  area: { id: string; nombre: string; codigo: string } | null;
    65→  perfilPuesto: { id: string; nombre_puesto: string; codigo: string; bandaSalarial: { nombre: string; salario_minimo: number; salario_maximo: number } | null } | null;
    66→  contratos: Array<{
    67→    id: string; tipo_contrato: string; salario_base_contrato: number; tipo_jornada: string;
    68→    fecha_inicio: string; fecha_fin: string | null; activo: boolean; observaciones: string | null;
    69→    perfilPuesto: { nombre_puesto: string } | null;
    70→  }>;
    71→  vacaciones: Array<{
    72→    id: string; anio: number; dias_derecho: number; dias_tomados: number;
    73→    dias_pendientes: number; dias_vendidos: number; estado: string;
    74→  }>;
    75→  documentos: Array<{
    76→    id: string; tipo_documento: string; nombre_archivo: string; descripcion: string | null; fecha_creacion: string;
    77→  }>;
    78→  incidencias: Array<{
    79→    id: string; tipo: string; estado: string; fecha_inicio: string; fecha_fin: string | null;
    80→    cantidad_horas: number | null; monto: number | null; descripcion: string | null;
    81→  }>;
    82→  cambios_salariales: Array<{
    83→    id: string; salario_anterior: number; salario_nuevo: number; tipo_cambio: string;
    84→    motivo: string | null; fecha_cambio: string;
    85→  }>;
    86→  usuario: { id: string; email: string; rol: string } | null;
    87→}
    88→
    89→export default function EmployeeDetail({ empleadoId, onBack, userRole, accessToken }: EmployeeDetailProps) {
    90→  const [empleado, setEmpleado] = useState<EmpleadoDetail | null>(null);
    91→  const [loading, setLoading] = useState(true);
    92→  const [editing, setEditing] = useState(false);
    93→  const [editData, setEditData] = useState<Record<string, unknown>>({});
    94→  const [saving, setSaving] = useState(false);
    95→  const [newContractOpen, setNewContractOpen] = useState(false);
    96→  const [contractData, setContractData] = useState({ tipo_contrato: 'INDEFINIDO', salario_base_contrato: '', tipo_jornada: 'COMPLETA', fecha_inicio: '', fecha_fin: '', observaciones: '' });
    97→  const { toast } = useToast();
    98→
    99→  const fetchEmpleado = useCallback(async () => {
   100→    setLoading(true);
   101→    try {
   102→      const res = await fetch(`/api/empleados/${empleadoId}`, {
   103→        headers: { Authorization: `Bearer ${accessToken}` },
   104→      });
   105→      const data = await res.json();
   106→      if (res.ok) {
   107→        setEmpleado(data.data);
   108→      }
   109→    } catch (err) {
   110→      console.error('Error fetching employee:', err);
   111→    } finally {
   112→      setLoading(false);
   113→    }
   114→  }, [empleadoId, accessToken]);
   115→
   116→  useEffect(() => { fetchEmpleado(); }, [fetchEmpleado]);
   117→
   118→  const canEdit = userRole === 'ADMIN' || userRole === 'ANALISTA';
   119→  const canEditOwn = userRole === 'EMPLEADO';
   120→
   121→  const formatDate = (d: string | null) => d ? new Date(d).toLocaleDateString('es-SV') : '—';
   122→  const formatSalary = (amount: number) => `$${amount.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`;
   123→  const getNombreCompleto = () => {
   124→    if (!empleado) return '';
   125→    return `${empleado.primer_nombre}${empleado.segundo_nombre ? ' ' + empleado.segundo_nombre : ''} ${empleado.primer_apellido}${empleado.segundo_apellido ? ' ' + empleado.segundo_apellido : ''}`;
   126→  };
   127→
   128→  const handleEdit = () => {
   129→    if (!empleado) return;
   130→    setEditData({
   131→      primer_nombre: empleado.primer_nombre,
   132→      segundo_nombre: empleado.segundo_nombre || '',
   133→      primer_apellido: empleado.primer_apellido,
   134→      segundo_apellido: empleado.segundo_apellido || '',
   135→      apellido_casada: empleado.apellido_casada || '',
   136→      dui: empleado.dui,
   137→      nit: empleado.nit || '',
   138→      genero: empleado.genero || '',
   139→      estado_civil: empleado.estado_civil || '',
   140→      direccion: empleado.direccion || '',
   141→      telefono: empleado.telefono || '',
   142→      email_personal: empleado.email_personal || '',
   143→      fecha_nacimiento: empleado.fecha_nacimiento ? new Date(empleado.fecha_nacimiento).toISOString().slice(0, 10) : '',
   144→      numero_isss: empleado.numero_isss || '',
   145→      numero_afp: empleado.numero_afp || '',
   146→      afp_administradora: empleado.afp_administradora || '',
   147→      tipo_sangre: empleado.tipo_sangre || '',
   148→      contacto_emergencia_nombre: empleado.contacto_emergencia_nombre || '',
   149→      contacto_emergencia_telefono: empleado.contacto_emergencia_telefono || '',
   150→      contacto_emergencia_relacion: empleado.contacto_emergencia_relacion || '',
   151→    });
   152→    setEditing(true);
   153→  };
   154→
   155→  const handleSave = async () => {
   156→    setSaving(true);
   157→    try {
   158→      const res = await fetch(`/api/empleados/${empleadoId}`, {
   159→        method: 'PUT',
   160→        headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${accessToken}` },
   161→        body: JSON.stringify(editData),
   162→      });
   163→      const data = await res.json();
   164→      if (res.ok) {
   165→        toast({ title: 'Empleado actualizado', description: 'Los datos han sido guardados correctamente' });
   166→        setEditing(false);
   167→        fetchEmpleado();
   168→      } else {
   169→        toast({ title: 'Error', description: data.error || 'Error al actualizar', variant: 'destructive' });
   170→      }
   171→    } catch {
   172→      toast({ title: 'Error', description: 'Error de conexión', variant: 'destructive' });
   173→    } finally {
   174→      setSaving(false);
   175→    }
   176→  };
   177→
   178→  const handleCreateContract = async () => {
   179→    setSaving(true);
   180→    try {
   181→      const res = await fetch(`/api/empleados/${empleadoId}/contratos`, {
   182→        method: 'POST',
   183→        headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${accessToken}` },
   184→        body: JSON.stringify({
   185→          ...contractData,
   186→          salario_base_contrato: parseFloat(contractData.salario_base_contrato),
   187→          fecha_fin: contractData.fecha_fin || null,
   188→        }),
   189→      });
   190→      const data = await res.json();
   191→      if (res.ok) {
   192→        toast({ title: 'Contrato creado', description: 'El nuevo contrato ha sido registrado' });
   193→        setNewContractOpen(false);
   194→        setContractData({ tipo_contrato: 'INDEFINIDO', salario_base_contrato: '', tipo_jornada: 'COMPLETA', fecha_inicio: '', fecha_fin: '', observaciones: '' });
   195→        fetchEmpleado();
   196→      } else {
   197→        toast({ title: 'Error', description: data.error || 'Error al crear contrato', variant: 'destructive' });
   198→      }
   199→    } catch {
   200→      toast({ title: 'Error', description: 'Error de conexión', variant: 'destructive' });
   201→    } finally {
   202→      setSaving(false);
   203→    }
   204→  };
   205→
   206→  if (loading) {
   207→    return (
   208→      <div className="space-y-4">
   209→        <Skeleton className="h-10 w-48" />
   210→        <Skeleton className="h-64 w-full" />
   211→      </div>
   212→    );
   213→  }
   214→
   215→  if (!empleado) {
   216→    return (
   217→      <Card className="shadow-sm">
   218→        <CardContent className="p-12 text-center">
   219→          <AlertCircle className="h-10 w-10 text-slate-400 mx-auto mb-2" />
   220→          <p className="text-slate-600">Empleado no encontrado</p>
   221→          <Button variant="outline" className="mt-4" onClick={onBack}>
   222→            <ArrowLeft className="h-4 w-4 mr-1" /> Volver
   223→          </Button>
   224→        </CardContent>
   225→      </Card>
   226→    );
   227→  }
   228→
   229→  const activeContract = empleado.contratos.find(c => c.activo);
   230→
   231→  const InfoField = ({ label, value }: { label: string; value: string | null | undefined }) => (
   232→    <div>
   233→      <p className="text-xs text-slate-500 font-medium">{label}</p>
   234→      <p className="text-sm text-slate-900 mt-0.5">{value || '—'}</p>
   235→    </div>
   236→  );
   237→
   238→  const EditField = ({ label, field, type = 'text' }: { label: string; field: string; type?: string }) => (
   239→    <div>
   240→      <Label className="text-xs">{label}</Label>
   241→      <Input
   242→        type={type}
   243→        value={String(editData[field] || '')}
   244→        onChange={e => setEditData(prev => ({ ...prev, [field]: e.target.value }))}
   245→        className="h-8 text-sm"
   246→      />
   247→    </div>
   248→  );
   249→
   250→  return (
   251→    <div className="space-y-4">
   252→      {/* Header */}
   253→      <div className="flex items-center gap-3">
   254→        <Button variant="ghost" size="sm" onClick={onBack}>
   255→          <ArrowLeft className="h-4 w-4 mr-1" /> Volver
   256→        </Button>
   257→        <div className="flex-1">
   258→          <h2 className="text-xl font-bold text-slate-900">{getNombreCompleto()}</h2>
   259→          <p className="text-sm text-slate-500">{empleado.codigo_empleado} • {empleado.area?.nombre || 'Sin área'}</p>
   260→        </div>
   261→        <Badge className={empleado.estado === 'ACTIVO' ? 'bg-emerald-100 text-emerald-700' : 'bg-slate-100 text-slate-600'}>
   262→          {empleado.estado}
   263→        </Badge>
   264→        {(canEdit || canEditOwn) && !editing && (
   265→          <Button variant="outline" size="sm" onClick={handleEdit}>
   266→            <Edit2 className="h-4 w-4 mr-1" /> Editar
   267→          </Button>
   268→        )}
   269→        {editing && (
   270→          <div className="flex gap-2">
   271→            <Button variant="outline" size="sm" onClick={() => setEditing(false)}><X className="h-4 w-4 mr-1" /> Cancelar</Button>
   272→            <Button size="sm" className="bg-emerald-600 hover:bg-emerald-700" onClick={handleSave} disabled={saving}>
   273→              {saving ? <Loader2 className="h-4 w-4 mr-1 animate-spin" /> : <Save className="h-4 w-4 mr-1" />} Guardar
   274→            </Button>
   275→          </div>
   276→        )}
   277→      </div>
   278→
   279→      {/* Tabs */}
   280→      <Tabs defaultValue="personal" className="space-y-4">
   281→        <TabsList className="grid grid-cols-3 lg:grid-cols-6 w-full h-auto gap-1">
   282→          <TabsTrigger value="personal" className="text-xs gap-1"><FileText className="h-3.5 w-3.5" /> Personal</TabsTrigger>
   283→          <TabsTrigger value="contrato" className="text-xs gap-1"><Briefcase className="h-3.5 w-3.5" /> Contrato</TabsTrigger>
   284→          <TabsTrigger value="previsional" className="text-xs gap-1"><Heart className="h-3.5 w-3.5" /> Previsional</TabsTrigger>
   285→          <TabsTrigger value="nomina" className="text-xs gap-1"><DollarSign className="h-3.5 w-3.5" /> Nómina</TabsTrigger>
   286→          <TabsTrigger value="vacaciones" className="text-xs gap-1"><Palmtree className="h-3.5 w-3.5" /> Vacaciones</TabsTrigger>
   287→          <TabsTrigger value="expediente" className="text-xs gap-1"><FolderOpen className="h-3.5 w-3.5" /> Expediente</TabsTrigger>
   288→        </TabsList>
   289→
   290→        {/* Tab 1: Datos Personales */}
   291→        <TabsContent value="personal">
   292→          <Card className="shadow-sm">
   293→            <CardHeader className="pb-3">
   294→              <CardTitle className="text-base">Datos Personales</CardTitle>
   295→            </CardHeader>
   296→            <CardContent>
   297→              {editing ? (
   298→                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   299→                  <EditField label="Primer Nombre" field="primer_nombre" />
   300→                  <EditField label="Segundo Nombre" field="segundo_nombre" />
   301→                  <EditField label="Primer Apellido" field="primer_apellido" />
   302→                  <EditField label="Segundo Apellido" field="segundo_apellido" />
   303→                  <EditField label="Apellido de Casada" field="apellido_casada" />
   304→                  <EditField label="DUI" field="dui" />
   305→                  <EditField label="NIT" field="nit" />
   306→                  <EditField label="Fecha de Nacimiento" field="fecha_nacimiento" type="date" />
   307→                  <div>
   308→                    <Label className="text-xs">Género</Label>
   309→                    <Select value={String(editData.genero || '')} onValueChange={v => setEditData(p => ({ ...p, genero: v }))}>
   310→                      <SelectTrigger className="h-8 text-sm"><SelectValue /></SelectTrigger>
   311→                      <SelectContent>
   312→                        <SelectItem value="">Seleccionar</SelectItem>
   313→                        <SelectItem value="MASCULINO">Masculino</SelectItem>
   314→                        <SelectItem value="FEMENINO">Femenino</SelectItem>
   315→                      </SelectContent>
   316→                    </Select>
   317→                  </div>
   318→                  <EditField label="Teléfono" field="telefono" />
   319→                  <EditField label="Email Personal" field="email_personal" type="email" />
   320→                  <div className="sm:col-span-2 lg:col-span-3">
   321→                    <EditField label="Dirección" field="direccion" />
   322→                  </div>
   323→                  <Separator className="sm:col-span-2 lg:col-span-3" />
   324→                  <EditField label="Contacto Emergencia Nombre" field="contacto_emergencia_nombre" />
   325→                  <EditField label="Contacto Emergencia Teléfono" field="contacto_emergencia_telefono" />
   326→                  <EditField label="Contacto Emergencia Relación" field="contacto_emergencia_relacion" />
   327→                </div>
   328→              ) : (
   329→                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   330→                  <InfoField label="Primer Nombre" value={empleado.primer_nombre} />
   331→                  <InfoField label="Segundo Nombre" value={empleado.segundo_nombre} />
   332→                  <InfoField label="Primer Apellido" value={empleado.primer_apellido} />
   333→                  <InfoField label="Segundo Apellido" value={empleado.segundo_apellido} />
   334→                  <InfoField label="Apellido de Casada" value={empleado.apellido_casada} />
   335→                  <InfoField label="DUI" value={empleado.dui} />
   336→                  <InfoField label="NIT" value={empleado.nit} />
   337→                  <InfoField label="Fecha de Nacimiento" value={formatDate(empleado.fecha_nacimiento)} />
   338→                  <InfoField label="Género" value={empleado.genero} />
   339→                  <InfoField label="Estado Civil" value={empleado.estado_civil} />
   340→                  <InfoField label="Nacionalidad" value={empleado.nacionalidad} />
   341→                  <InfoField label="Teléfono" value={empleado.telefono} />
   342→                  <InfoField label="Email Personal" value={empleado.email_personal} />
   343→                  <div className="sm:col-span-2 lg:col-span-3"><InfoField label="Dirección" value={empleado.direccion} /></div>
   344→                  <Separator className="sm:col-span-2 lg:col-span-3" />
   345→                  <InfoField label="Contacto Emergencia" value={empleado.contacto_emergencia_nombre} />
   346→                  <InfoField label="Teléfono Emergencia" value={empleado.contacto_emergencia_telefono} />
   347→                  <InfoField label="Relación Emergencia" value={empleado.contacto_emergencia_relacion} />
   348→                </div>
   349→              )}
   350→            </CardContent>
   351→          </Card>
   352→        </TabsContent>
   353→
   354→        {/* Tab 2: Contrato */}
   355→        <TabsContent value="contrato">
   356→          <div className="space-y-4">
   357→            {/* Active Contract */}
   358→            <Card className="shadow-sm">
   359→              <CardHeader className="pb-3">
   360→                <div className="flex items-center justify-between">
   361→                  <CardTitle className="text-base">Contrato Vigente</CardTitle>
   362→                  {canEdit && (
   363→                    <Button size="sm" variant="outline" onClick={() => setNewContractOpen(true)}>
   364→                      <Plus className="h-4 w-4 mr-1" /> Nuevo Contrato
   365→                    </Button>
   366→                  )}
   367→                </div>
   368→              </CardHeader>
   369→              <CardContent>
   370→                {activeContract ? (
   371→                  <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   372→                    <InfoField label="Tipo de Contrato" value={activeContract.tipo_contrato} />
   373→                    <InfoField label="Salario Base" value={formatSalary(activeContract.salario_base_contrato)} />
   374→                    <InfoField label="Jornada" value={activeContract.tipo_jornada} />
   375→                    <InfoField label="Fecha Inicio" value={formatDate(activeContract.fecha_inicio)} />
   376→                    <InfoField label="Fecha Fin" value={activeContract.fecha_fin ? formatDate(activeContract.fecha_fin) : 'Indefinido'} />
   377→                    <InfoField label="Puesto" value={activeContract.perfilPuesto?.nombre_puesto || '—'} />
   378→                    <div className="sm:col-span-2 lg:col-span-3"><InfoField label="Observaciones" value={activeContract.observaciones} /></div>
   379→                  </div>
   380→                ) : (
   381→                  <p className="text-sm text-slate-500 text-center py-8">No hay contrato vigente</p>
   382→                )}
   383→              </CardContent>
   384→            </Card>
   385→
   386→            {/* Contract History */}
   387→            <Card className="shadow-sm">
   388→              <CardHeader className="pb-3">
   389→                <CardTitle className="text-base">Historial de Contratos</CardTitle>
   390→              </CardHeader>
   391→              <CardContent className="p-0">
   392→                <Table>
   393→                  <TableHeader>
   394→                    <TableRow>
   395→                      <TableHead>Tipo</TableHead>
   396→                      <TableHead>Salario</TableHead>
   397→                      <TableHead>Inicio</TableHead>
   398→                      <TableHead>Fin</TableHead>
   399→                      <TableHead>Estado</TableHead>
   400→                    </TableRow>
   401→                  </TableHeader>
   402→                  <TableBody>
   403→                    {empleado.contratos.length === 0 ? (
   404→                      <TableRow><TableCell colSpan={5} className="text-center py-6 text-slate-500">Sin contratos registrados</TableCell></TableRow>
   405→                    ) : (
   406→                      empleado.contratos.map(c => (
   407→                        <TableRow key={c.id}>
   408→                          <TableCell className="text-sm">{c.tipo_contrato}</TableCell>
   409→                          <TableCell className="text-sm font-mono">{formatSalary(c.salario_base_contrato)}</TableCell>
   410→                          <TableCell className="text-sm">{formatDate(c.fecha_inicio)}</TableCell>
   411→                          <TableCell className="text-sm">{c.fecha_fin ? formatDate(c.fecha_fin) : 'Indefinido'}</TableCell>
   412→                          <TableCell>
   413→                            <Badge variant={c.activo ? 'default' : 'secondary'} className={c.activo ? 'bg-emerald-100 text-emerald-700' : ''}>
   414→                              {c.activo ? 'Vigente' : 'Finalizado'}
   415→                            </Badge>
   416→                          </TableCell>
   417→                        </TableRow>
   418→                      ))
   419→                    )}
   420→                  </TableBody>
   421→                </Table>
   422→              </CardContent>
   423→            </Card>
   424→          </div>
   425→        </TabsContent>
   426→
   427→        {/* Tab 3: Datos Previsionales */}
   428→        <TabsContent value="previsional">
   429→          <Card className="shadow-sm">
   430→            <CardHeader className="pb-3">
   431→              <CardTitle className="text-base">Datos Previsionales</CardTitle>
   432→            </CardHeader>
   433→            <CardContent>
   434→              <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   435→                <InfoField label="Número ISSS" value={empleado.numero_isss} />
   436→                <InfoField label="Número AFP (NUP)" value={empleado.numero_afp} />
