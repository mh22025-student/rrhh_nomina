   308→  const [lockoutCountdown, setLockoutCountdown] = useState('');
   309→
   310→  // Lockout timer effect
   311→  useEffect(() => {
   312→    if (!lockoutUntil) return;
   313→    const interval = setInterval(() => {
   314→      const remaining = lockoutUntil - Date.now();
   315→      if (remaining <= 0) {
   316→        setLockoutUntil(null);
   317→        setFailedAttempts(0);
   318→        setLockoutCountdown('');
   319→        clearInterval(interval);
   320→        return;
   321→      }
   322→      const mins = Math.floor(remaining / 60000);
   323→      const secs = Math.floor((remaining % 60000) / 1000);
   324→      setLockoutCountdown(`${mins}:${secs.toString().padStart(2, '0')}`);

   334→    try {
   335→      await onLogin(email, password);
   336→    } catch (err) {
   337→      const newAttempts = failedAttempts + 1;
   338→      setFailedAttempts(newAttempts);
   339→      const errorMsg = err instanceof Error ? err.message : 'Error desconocido';

   341→      // Lockout after 5 failed attempts
