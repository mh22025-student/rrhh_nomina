   307→    if (!lockoutUntil) return;
   308→    const interval = setInterval(() => {
   309→      const remaining = lockoutUntil - Date.now();
   310→      if (remaining <= 0) {
   311→        setLockoutUntil(null);
   312→        setFailedAttempts(0);
   313→        setLockoutCountdown('');
   314→        clearInterval(interval);
   315→        return;
   316→      }
   317→      const mins = Math.floor(remaining / 60000);
   318→      const secs = Math.floor((remaining % 60000) / 1000);
   319→      setLockoutCountdown(`${mins}:${secs.toString().padStart(2, '0')}`);

   330→      await onLogin(email, password);

   654→// ============================================================
   655→// PASSWORD RECOVERY DIALOG
   656→// ============================================================
   657→interface PasswordRecoveryDialogProps {
   658→  open: boolean;
   659→  onOpenChange: (open: boolean) => void;
   660→}
   661→
   662→function PasswordRecoveryDialog({ open, onOpenChange }: PasswordRecoveryDialogProps) {
   663→  const [step, setStep] = useState(1);
   664→  const [email, setEmail] = useState('');
   665→  const [otp, setOtp] = useState('');
   666→  const [otpId, setOtpId] = useState('');
   667→  const [newPassword, setNewPassword] = useState('');
   668→  const [confirmPassword, setConfirmPassword] = useState('');
   669→  const [demoOtp, setDemoOtp] = useState<string | null>(null);
   670→  const [isLoading, setIsLoading] = useState(false);
   671→  const [error, setError] = useState<string | null>(null);
   672→  const [success, setSuccess] = useState<string | null>(null);
   673→  const { toast } = useToast();
   674→
   675→  const handleReset = () => {
   676→    setStep(1);
   677→    setEmail('');
   678→    setOtp('');
   679→    setOtpId('');
   680→    setNewPassword('');
   681→    setConfirmPassword('');
   682→    setDemoOtp(null);
   683→    setIsLoading(false);
   684→    setError(null);
   685→    setSuccess(null);
   686→  };
   687→
   688→  const handleClose = (newOpen: boolean) => {
   689→    if (!newOpen) handleReset();
   690→    onOpenChange(newOpen);
   691→  };
   692→
   693→  const handleStep1 = async () => {
   694→    setError(null);
   695→    setIsLoading(true);
   696→    try {
   697→      const res = await fetch('/api/auth/forgot-password', {
   698→        method: 'POST',
   699→        headers: { 'Content-Type': 'application/json' },
   700→        body: JSON.stringify({ email }),
   701→      });
   702→      const data = await res.json();
   703→      if (!res.ok) {
   704→        setError(data.error || 'Error al solicitar código');
   705→        return;
   706→      }
   707→      setDemoOtp(data.demo_otp);
   708→      setSuccess('Código enviado. Verifique su correo.');
   709→      setStep(2);
   710→    } catch {
   711→      setError('Error de conexión');
   712→    } finally {
   713→      setIsLoading(false);
   714→    }
   715→  };
   716→
   717→  const handleStep2 = async () => {
   718→    setError(null);
   719→    setIsLoading(true);
   720→    try {
   721→      const res = await fetch('/api/auth/verify-otp', {
   722→        method: 'POST',
   723→        headers: { 'Content-Type': 'application/json' },
   724→        body: JSON.stringify({ email, otp }),
   725→      });
   726→      const data = await res.json();
   727→      if (!res.ok) {
   728→        setError(data.error || 'Código inválido');
   729→        return;
   730→      }
   731→      setOtpId(data.otp_id);
   732→      setSuccess('Código verificado');
   733→      setStep(3);
   734→    } catch {
   735→      setError('Error de conexión');
   736→    } finally {
   737→      setIsLoading(false);
   738→    }
   739→  };
   740→
   741→  const handleStep3 = async () => {
   742→    setError(null);
   743→    if (newPassword !== confirmPassword) {
   744→      setError('Las contraseñas no coinciden');
   745→      return;
   746→    }
   747→    if (newPassword.length < 8) {
   748→      setError('La contraseña debe tener al menos 8 caracteres');
   749→      return;
   750→    }
   751→    setIsLoading(true);
   752→    try {
   753→      const res = await fetch('/api/auth/reset-password', {
   754→        method: 'POST',
   755→        headers: { 'Content-Type': 'application/json' },
   756→        body: JSON.stringify({ otp_id: otpId, email, new_password: newPassword }),
   757→      });
   758→      const data = await res.json();
   759→      if (!res.ok) {
   760→        setError(data.error || 'Error al actualizar contraseña');
   761→        return;
   762→      }
   763→      toast({
   764→        title: 'Contraseña actualizada',
   765→        description: 'Puede iniciar sesión con su nueva contraseña',
   766→      });
   767→      handleClose(false);
   768→    } catch {
   769→      setError('Error de conexión');
   770→    } finally {
   771→      setIsLoading(false);
   772→    }
   773→  };
   774→
   775→  const stepLabels = ['Email', 'Verificación', 'Nueva Contraseña'];
   776→  const progressValue = (step / 3) * 100;
   777→
   778→  return (
   779→    <Dialog open={open} onOpenChange={handleClose}>
   780→      <DialogContent className="sm:max-w-md">
   781→        <DialogHeader>
   782→          <DialogTitle className="flex items-center gap-2">
   783→            <Lock className="h-5 w-5 text-emerald-600" />
   784→            Recuperar Contraseña
   785→          </DialogTitle>
   786→          <DialogDescription>
   787→            Siga los pasos para restablecer su contraseña
   788→          </DialogDescription>
   789→        </DialogHeader>
   790→
   791→        {/* Progress indicator */}
   792→        <div className="space-y-2">
   793→          <div className="flex justify-between text-xs text-slate-500">
   794→            {stepLabels.map((label, i) => (
   795→              <span key={i} className={step > i ? 'text-emerald-600 font-medium' : step === i + 1 ? 'text-slate-900 font-medium' : ''}>
   796→                {i + 1}. {label}
   797→              </span>
   798→            ))}
   799→          </div>
   800→          <Progress value={progressValue} className="h-2" />
   801→        </div>
   802→
   803→        <div className="space-y-4">
   804→          {error && (
   805→            <div className="flex items-center gap-2 text-sm text-red-600 bg-red-50 p-3 rounded-lg">
   806→              <AlertCircle className="h-4 w-4 shrink-0" />
   807→              <span>{error}</span>
   808→            </div>
   809→          )}
   810→
   811→          {success && step > 1 && (
   812→            <div className="flex items-center gap-2 text-sm text-emerald-700 bg-emerald-50 p-3 rounded-lg">
   813→              <CheckCircle className="h-4 w-4 shrink-0" />
   814→              <span>{success}</span>
   815→            </div>
   816→          )}
   817→
   818→          {/* Step 1: Email */}
   819→          {step === 1 && (
   820→            <div className="space-y-4">
   821→              <div className="space-y-2">
   822→                <Label>Correo Electrónico</Label>
   823→                <Input
   824→                  type="email"
   825→                  placeholder="usuario@nomina.gob.sv"
   826→                  value={email}
   827→                  onChange={(e) => setEmail(e.target.value)}
   828→                  disabled={isLoading}
   829→                />
   830→              </div>
   831→              <Button onClick={handleStep1} disabled={isLoading || !email} className="w-full bg-emerald-600 hover:bg-emerald-700">
   832→                {isLoading ? <Loader2 className="mr-2 h-4 w-4 animate-spin" /> : null}
   833→                Enviar Código
   834→              </Button>
   835→            </div>
   836→          )}
   837→
   838→          {/* Step 2: OTP */}
   839→          {step === 2 && (
   840→            <div className="space-y-4">
   841→              <div className="space-y-2">
   842→                <Label>Código de Verificación</Label>
   843→                <Input
   844→                  placeholder="000000"
   845→                  value={otp}
   846→                  onChange={(e) => setOtp(e.target.value.replace(/\D/g, '').slice(0, 6))}
   847→                  disabled={isLoading}
   848→                  className="text-center text-lg tracking-[0.5em] font-mono"
   849→                  maxLength={6}
   850→                />
   851→              </div>
   852→              {demoOtp && (
   853→                <p className="text-xs text-amber-700 bg-amber-50 p-2 rounded">
   854→                  Demo - Código: <strong>{demoOtp}</strong>
   855→                </p>
   856→              )}
   857→              <div className="flex gap-2">
   858→                <Button variant="outline" onClick={() => setStep(1)} className="flex-1">
   859→                  <ArrowLeft className="mr-1 h-4 w-4" /> Volver
   860→                </Button>
   861→                <Button onClick={handleStep2} disabled={isLoading || otp.length !== 6} className="flex-1 bg-emerald-600 hover:bg-emerald-700">
   862→                  {isLoading ? <Loader2 className="mr-2 h-4 w-4 animate-spin" /> : null}
   863→                  Verificar
   864→                </Button>
   865→              </div>
   866→            </div>
   867→          )}
   868→
   869→          {/* Step 3: New Password */}
   870→          {step === 3 && (
   871→            <div className="space-y-4">
   872→              <div className="space-y-2">
   873→                <Label>Nueva Contraseña</Label>
   874→                <Input
   875→                  type="password"
   876→                  placeholder="Mínimo 8 caracteres"
   877→                  value={newPassword}
   878→                  onChange={(e) => setNewPassword(e.target.value)}
   879→                  disabled={isLoading}
   880→                />
   881→              </div>
   882→              <div className="space-y-2">
   883→                <Label>Confirmar Contraseña</Label>
   884→                <Input
   885→                  type="password"
   886→                  placeholder="Repita la contraseña"
