   214→function CalendarView({ incidencias, onDayClick }: { incidencias: Incidencia[]; onDayClick: (day: number) => void }) {
   215→  const [currentMonth, setCurrentMonth] = useState(() => {
   216→    const now = new Date();
   217→    return { year: now.getFullYear(), month: now.getMonth() };
   218→  });
   219→
   220→  const daysInMonth = new Date(currentMonth.year, currentMonth.month + 1, 0).getDate();
   221→  const firstDayOfWeek = new Date(currentMonth.year, currentMonth.month, 1).getDay();
   222→  const monthName = new Date(currentMonth.year, currentMonth.month).toLocaleString('es-SV', { month: 'long', year: 'numeric' });
   223→
   224→  const incidencesByDay = useMemo(() => {
   225→    const map: Record<number, Incidencia[]> = {};
   226→    incidencias.forEach(inc => {
   227→      const d = new Date(inc.fecha_inicio);
   228→      if (d.getFullYear() === currentMonth.year && d.getMonth() === currentMonth.month) {
   229→        const day = d.getDate();
   230→        if (!map[day]) map[day] = [];
   231→        map[day].push(inc);
   232→      }
   233→    });
   234→    return map;
   235→  }, [incidencias, currentMonth]);
   236→
   237→  const dayNames = ['Dom', 'Lun', 'Mar', 'Mié', 'Jue', 'Vie', 'Sáb'];
   238→
   239→  const prevMonth = () => {
   240→    setCurrentMonth(p => p.month === 0 ? { year: p.year - 1, month: 11 } : { ...p, month: p.month - 1 });
   241→  };
   242→  const nextMonth = () => {
   243→    setCurrentMonth(p => p.month === 11 ? { year: p.year + 1, month: 0 } : { ...p, month: p.month + 1 });
   244→  };
   245→
   246→  const cells: (number | null)[] = [];
   247→  for (let i = 0; i < firstDayOfWeek; i++) cells.push(null);
   248→  for (let d = 1; d <= daysInMonth; d++) cells.push(d);
   249→
   250→  return (
   251→    <Card className="shadow-sm dark:bg-slate-900 dark:border-slate-800">
   252→      <CardContent className="p-4">
   253→        <div className="flex items-center justify-between mb-4">
   254→          <Button variant="ghost" size="sm" onClick={prevMonth} className="h-8 w-8 p-0 dark:text-slate-300">
   255→            <ChevronLeft className="h-4 w-4" />
   256→          </Button>
   257→          <h3 className="text-sm font-semibold capitalize text-slate-900 dark:text-slate-100">{monthName}</h3>
   258→          <Button variant="ghost" size="sm" onClick={nextMonth} className="h-8 w-8 p-0 dark:text-slate-300">
   259→            <ChevronRight className="h-4 w-4" />
   260→          </Button>
   261→        </div>
   262→        {/* Day headers */}
   263→        <div className="grid grid-cols-7 gap-1 mb-1">
   264→          {dayNames.map(d => (
   265→            <div key={d} className="text-center text-[10px] font-medium text-slate-400 dark:text-slate-500 py-1">{d}</div>
   266→          ))}
   267→        </div>
   268→        {/* Calendar grid */}
   269→        <div className="grid grid-cols-7 gap-1">
   270→          {cells.map((day, idx) => {
   271→            if (day === null) return <div key={`empty-${idx}`} className="h-16" />;
   272→            const dayIncs = incidencesByDay[day] || [];
   273→            const today = new Date();
   274→            const isToday = day === today.getDate() && currentMonth.month === today.getMonth() && currentMonth.year === today.getFullYear();
   275→            return (
   276→              <button
   277→                key={day}
   278→                onClick={() => dayIncs.length > 0 && onDayClick(day)}
   279→                className={`h-16 rounded-lg border text-left p-1.5 transition-all duration-200 relative ${
   280→                  isToday
   281→                    ? 'border-emerald-400 dark:border-emerald-600 bg-emerald-50/50 dark:bg-emerald-900/20'
   282→                    : 'border-slate-100 dark:border-slate-800 hover:border-slate-300 dark:hover:border-slate-600 hover:bg-slate-50 dark:hover:bg-slate-800'
   283→                } ${dayIncs.length > 0 ? 'cursor-pointer' : 'cursor-default'}`}
   284→              >
   285→                <span className={`text-[11px] font-medium ${isToday ? 'text-emerald-600 dark:text-emerald-400' : 'text-slate-600 dark:text-slate-400'}`}>{day}</span>
   286→                {dayIncs.length > 0 && (
   287→                  <div className="flex flex-wrap gap-0.5 mt-0.5">
   288→                    {dayIncs.slice(0, 4).map((inc, i) => (
   289→                      <span key={i} className={`w-2 h-2 rounded-full ${TIPO_COLORS[inc.tipo]?.dot || 'bg-slate-400'}`} title={TIPO_LABELS[inc.tipo] || inc.tipo} />
   290→                    ))}
   291→                    {dayIncs.length > 4 && (
   292→                      <span className="text-[8px] text-slate-400 dark:text-slate-500 leading-none">+{dayIncs.length - 4}</span>
   293→                    )}
   294→                  </div>
   295→                )}
   296→              </button>
   297→            );
   298→          })}
   299→        </div>
   300→        {/* Legend */}
   301→        <div className="flex flex-wrap gap-3 mt-3 pt-3 border-t border-slate-100 dark:border-slate-800">
   302→          {Object.entries(TIPO_LABELS).slice(0, 5).map(([key, label]) => (
   303→            <div key={key} className="flex items-center gap-1.5">
   304→              <span className={`w-2.5 h-2.5 rounded-full ${TIPO_COLORS[key]?.dot}`} />
   305→              <span className="text-[10px] text-slate-500 dark:text-slate-400">{label}</span>
   306→            </div>
   307→          ))}
   308→        </div>
   309→      </CardContent>
   310→    </Card>
   311→  );
   312→}

   316→function StatisticsPanel({ incidencias }: { incidencias: Incidencia[] }) {
   317→  const stats = useMemo(() => {
   318→    const byType: Record<string, number> = {};
   319→    incidencias.forEach(inc => {
   320→      byType[inc.tipo] = (byType[inc.tipo] || 0) + 1;
   321→    });
   322→
   323→    const byMonth: Record<string, number> = {};
   324→    incidencias.forEach(inc => {
   325→      const d = new Date(inc.fecha_inicio);
   326→      const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   327→      byMonth[key] = (byMonth[key] || 0) + 1;
   328→    });
   329→
   330→    const sortedMonths = Object.entries(byMonth).sort((a, b) => a[0].localeCompare(b[0])).slice(-6);
   331→
   332→    const approved = incidencias.filter(i => i.estado === 'APROBADA').length;
   333→    const total = incidencias.length;
   334→    const approvalRate = total > 0 ? Math.round((approved / total) * 100) : 0;
   335→
   336→    let totalProcessingHours = 0;
   337→    let processedCount = 0;
   338→    incidencias.forEach(inc => {
   339→      if (inc.fecha_creacion && inc.fecha_actualizacion && inc.estado !== 'PENDIENTE') {
   340→        const created = new Date(inc.fecha_creacion).getTime();
   341→        const updated = new Date(inc.fecha_actualizacion).getTime();
   342→        const hours = (updated - created) / (1000 * 60 * 60);
   343→        if (hours >= 0) {
   344→          totalProcessingHours += hours;
   345→          processedCount++;
   346→        }
   347→      }
   348→    });
   349→    const avgProcessingHours = processedCount > 0 ? totalProcessingHours / processedCount : 0;
   350→
   351→    return { byType, sortedMonths, approvalRate, avgProcessingHours, total, approved };
   352→  }, [incidencias]);
   353→
   354→  const totalForPie = Object.values(stats.byType).reduce((s, v) => s + v, 0);
   355→
   356→  // Build conic-gradient for pie chart
   357→  const conicGradient = useMemo(() => {
   358→    if (totalForPie === 0) return 'conic-gradient(#e2e8f0 0deg 360deg)';
   359→    let currentDeg = 0;
   360→    const stops: string[] = [];
   361→    Object.entries(stats.byType).forEach(([tipo, count]) => {
   362→      const deg = (count / totalForPie) * 360;
   363→      const color = TIPO_COLORS[tipo]?.hex || '#94a3b8';
   364→      stops.push(`${color} ${currentDeg}deg ${currentDeg + deg}deg`);
   365→      currentDeg += deg;
   366→    });
   367→    return `conic-gradient(${stops.join(', ')})`;
   368→  }, [stats.byType, totalForPie]);
   369→
   370→  const maxMonthCount = Math.max(...stats.sortedMonths.map(([, c]) => c), 1);
   371→
   372→  return (
   373→    <Card className="shadow-sm dark:bg-slate-900 dark:border-slate-800">
   374→      <CardHeader className="pb-3">
   375→        <CardTitle className="text-sm font-semibold text-slate-900 dark:text-slate-100 flex items-center gap-2">
   376→          <BarChart3 className="h-4 w-4 text-emerald-600 dark:text-emerald-400" />
   377→          Estadísticas de Incidencias
   378→        </CardTitle>
   379→      </CardHeader>
   380→      <CardContent className="p-4 pt-0">
   381→        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
   382→          {/* Pie chart - Distribution by type */}
   383→          <div className="flex flex-col items-center">
   384→            <p className="text-[10px] font-semibold text-slate-500 dark:text-slate-400 uppercase tracking-wider mb-2 flex items-center gap-1">
   385→              <PieChart className="h-3 w-3" /> Por Tipo
   386→            </p>
   387→            <div className="relative w-28 h-28 rounded-full" style={{ background: conicGradient }}>
   388→              <div className="absolute inset-3 rounded-full bg-white dark:bg-slate-900 flex items-center justify-center">
   389→                <span className="text-lg font-bold text-slate-900 dark:text-slate-100">{totalForPie}</span>
   390→              </div>
   391→            </div>
   392→            <div className="flex flex-wrap gap-x-3 gap-y-1 mt-2 justify-center">
   393→              {Object.entries(stats.byType).map(([tipo, count]) => (
   394→                <div key={tipo} className="flex items-center gap-1">
   395→                  <span className={`w-2 h-2 rounded-full ${TIPO_COLORS[tipo]?.dot}`} />
   396→                  <span className="text-[9px] text-slate-500 dark:text-slate-400">{TIPO_LABELS[tipo]} ({count})</span>
   397→                </div>
   398→              ))}
   399→            </div>
   400→          </div>
   401→
   402→          {/* Bar chart - Monthly trend */}
   403→          <div className="flex flex-col">
   404→            <p className="text-[10px] font-semibold text-slate-500 dark:text-slate-400 uppercase tracking-wider mb-2 flex items-center gap-1">
   405→              <TrendingUp className="h-3 w-3" /> Tendencia Mensual
   406→            </p>
   407→            <div className="flex items-end gap-1.5 h-24">
   408→              {stats.sortedMonths.map(([month, count]) => {

   454→          </div>
