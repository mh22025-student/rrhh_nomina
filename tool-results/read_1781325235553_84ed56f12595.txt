    95→interface DocItem {
    96→  id: string;
    97→  title: string;
    98→  description: string;

   109→
   110→function getDocStatusColor(status: string): string {
   111→  switch (status) {
   112→    case 'Vigente': return 'bg-emerald-100 text-emerald-700 dark:bg-emerald-900/40 dark:text-emerald-300';
   113→    case 'Expirado': return 'bg-red-100 text-red-700 dark:bg-red-900/40 dark:text-red-300';
   114→    case 'Borrador': return 'bg-amber-100 text-amber-700 dark:bg-amber-900/40 dark:text-amber-300';
   115→    default: return 'bg-slate-100 text-slate-700 dark:bg-slate-800 dark:text-slate-300';
   116→  }
   117→}
   118→
   119→function getContractStatus(fechaFin: string | null, activo: boolean): 'Vigente' | 'Expirado' | 'Borrador' {
   120→  if (!activo) return 'Expirado';
   121→  if (!fechaFin) return 'Vigente';
   122→  return new Date(fechaFin) > new Date() ? 'Vigente' : 'Expirado';
   123→}

   649→function DocumentGenerator({
   650→  docGenType,
   651→  setDocGenType,
   652→  empleado,
   653→  planillas,
   654→  selectedPlanilla,
   655→  setSelectedPlanilla,
   656→  aguinaldoAnio,

   662→  addToRecentDocs,
   663→  docGenGenerating,
   664→  setDocGenGenerating,
   665→  onClose,
   666→  toast,
   667→}: {
   668→  docGenType: string;
   669→  setDocGenType: (v: string) => void;
   670→  empleado: EmpleadoDetail;
   671→  planillas: Array<{ id: string; codigo_planilla: string; tipo: string; estado: string; fecha_inicio_periodo: string; fecha_fin_periodo: string }>;
   672→  selectedPlanilla: string;
   673→  setSelectedPlanilla: (v: string) => void;
   674→  aguinaldoAnio: string;
   675→  setAguinaldoAnio: (v: string) => void;
   676→  accessToken: string | null;
   677→  formatDate: (d: string | null) => string;
   678→  formatSalary: (n: number) => string;
   679→  getNombreCompleto: () => string;
   680→  addToRecentDocs: (doc: { id: string; title: string; type: string; date: string; category: string }) => void;
   681→  docGenGenerating: boolean;
   682→  setDocGenGenerating: (v: boolean) => void;
   683→  onClose: () => void;
   684→  toast: (opts: { title: string; description?: string; variant?: string }) => void;
   685→}) {
   686→  const nombre = getNombreCompleto();
   687→  const activeContract = empleado.contratos.find(c => c.activo);
   688→
   689→  const handleGenerate = async () => {
   690→    setDocGenGenerating(true);
   691→    try {
   692→      if (docGenType === 'boleta_pago') {
   693→        if (!selectedPlanilla) {
   694→          toast({ title: 'Error', description: 'Seleccione una planilla', variant: 'destructive' });
   695→          setDocGenGenerating(false);
   696→          return;
   697→        }
   698→        const p = planillas.find(pl => pl.id === selectedPlanilla);
   699→        const res = await fetch(`/api/nomina/planillas/${selectedPlanilla}/boleta?empleado_id=${empleado.id}`, {
   700→          headers: { Authorization: `Bearer ${accessToken}` },
   701→        });
   702→        if (!res.ok) {
   703→          const data = await res.json().catch(() => ({}));
   704→          toast({ title: 'Error', description: (data as { error?: string }).error || 'Error al generar boleta', variant: 'destructive' });
   705→          setDocGenGenerating(false);
   706→          return;
   707→        }
   708→        const blob = await res.blob();
   709→        const url = window.URL.createObjectURL(blob);
   710→        const a = document.createElement('a');
   711→        a.href = url;
   712→        a.download = `boleta-${empleado.codigo_empleado}-${p?.codigo_planilla || 'planilla'}.pdf`;
   713→        a.click();
   714→        window.URL.revokeObjectURL(url);
   715→        addToRecentDocs({
   716→          id: `boleta-${selectedPlanilla}`,
   717→          title: `Boleta ${p?.codigo_planilla || ''}`,
   718→          type: 'Boleta de Pago',
   719→          date: new Date().toISOString(),
   720→          category: 'boletas',
   721→        });
   722→      } else if (docGenType === 'constancia_aguinaldo') {
   723→        const res = await fetch(`/api/nomina/aguinaldo/pdf?empleado_id=${empleado.id}&anio=${aguinaldoAnio}`, {
   724→          headers: { Authorization: `Bearer ${accessToken}` },
   725→        });
   726→        if (!res.ok) {
   727→          const data = await res.json().catch(() => ({}));
   728→          toast({ title: 'Error', description: (data as { error?: string }).error || 'Error al generar constancia', variant: 'destructive' });
   729→          setDocGenGenerating(false);
   730→          return;
   731→        }
   732→        const blob = await res.blob();
   733→        const url = window.URL.createObjectURL(blob);
   734→        const a = document.createElement('a');
   735→        a.href = url;
   736→        a.download = `aguinaldo-${empleado.codigo_empleado}-${aguinaldoAnio}.pdf`;
   737→        a.click();
   738→        window.URL.revokeObjectURL(url);
   739→        addToRecentDocs({
   740→          id: `aguinaldo-${aguinaldoAnio}`,
   741→          title: `Constancia Aguinaldo ${aguinaldoAnio}`,
   742→          type: 'Constancia de Aguinaldo',
   743→          date: new Date().toISOString(),
   744→          category: 'constancias',
   745→        });
   746→      } else {
   747→        // Text-based documents
   748→        let content = '';
   749→        let filename = '';
   750→        let title = '';
   751→        let category = 'constancias';
   752→
   753→        if (docGenType === 'constancia_empleo') {
   754→          content = generateConstanciaEmpleoHTML(empleado, formatDate, formatSalary);
   755→          filename = `constancia-empleo-${empleado.codigo_empleado}.txt`;

   764→          title = 'Carta de Referencia';
   765→          category = 'cartas';
   766→        }
   767→
   768→        const blob = new Blob([content], { type: 'text/plain;charset=utf-8' });
   769→        const url = window.URL.createObjectURL(blob);
   770→        const a = document.createElement('a');
   771→        a.href = url;
   772→        a.download = filename;
   773→        a.click();
   774→        window.URL.revokeObjectURL(url);
   775→        addToRecentDocs({
   776→          id: docGenType,
   777→          title,
   778→          type: title,
   779→          date: new Date().toISOString(),
   780→          category,

   795→      return (

   943→/* ─── circular progress ring ─── */
   944→function CircularProgress({ value, max, size = 80, strokeWidth = 6, color = 'emerald' }: {
   945→  value: number; max: number; size?: number; strokeWidth?: number; color?: string;
   946→}) {
   947→  const radius = (size - strokeWidth) / 2;
   948→  const circumference = radius * 2 * Math.PI;
   949→  const pct = max > 0 ? Math.min(value / max, 1) : 0;
   950→  const offset = circumference - pct * circumference;
   951→  const colorMap: Record<string, string> = {
   952→    emerald: 'text-emerald-500',
   953→    amber: 'text-amber-500',
   954→    sky: 'text-sky-500',
   955→    purple: 'text-purple-500',
   956→  };
   957→  const strokeClass = colorMap[color] || colorMap.emerald;
   958→  return (
   959→    <svg width={size} height={size} className="-rotate-90">
   960→      <circle cx={size / 2} cy={size / 2} r={radius} fill="none" stroke="currentColor" strokeWidth={strokeWidth} className="text-slate-200 dark:text-slate-700" />
   961→      <circle cx={size / 2} cy={size / 2} r={radius} fill="none" stroke="currentColor" strokeWidth={strokeWidth} strokeDasharray={circumference} strokeDashoffset={offset} strokeLinecap="round" className={`${strokeClass} transition-all duration-700`} />
   962→    </svg>
   963→  );
   964→}
