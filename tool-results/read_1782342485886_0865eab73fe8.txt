    59→const fmt = (n: number) => `$${n.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`;
    60→const fmtShort = (n: number) => {
    61→  if (n >= 1000000) return `$${(n / 1000000).toFixed(1)}M`;
    62→  if (n >= 1000) return `$${(n / 1000).toFixed(1)}K`;
    63→  return `$${n.toFixed(0)}`;

    65→const fmtDate = (d: string) => {
    66→  try {
    67→    const date = new Date(d);
    68→    return date.toLocaleDateString('es-SV', { day: '2-digit', month: '2-digit', year: 'numeric' });
    69→  } catch {
    70→    return d;
    71→  }
    72→};
    73→const fmtDateTime = (d: string) => {
    74→  try {
    75→    const date = new Date(d);
    76→    return date.toLocaleDateString('es-SV', { day: '2-digit', month: 'short', year: 'numeric' }) +
    77→      ' ' + date.toLocaleTimeString('es-SV', { hour: '2-digit', minute: '2-digit' });
    78→  } catch {
    79→    return d;
    80→  }
    81→};
    82→const fmtRelativeTime = (d: string) => {
    83→  try {
    84→    const date = new Date(d);
    85→    const now = new Date();
    86→    const diffMs = now.getTime() - date.getTime();
    87→    const diffMins = Math.floor(diffMs / 60000);
    88→    if (diffMins < 1) return 'Justo ahora';
    89→    if (diffMins < 60) return `Hace ${diffMins} min`;
    90→    const diffHours = Math.floor(diffMins / 60);
    91→    if (diffHours < 24) return `Hace ${diffHours}h`;
    92→    const diffDays = Math.floor(diffHours / 24);
    93→    if (diffDays < 7) return `Hace ${diffDays}d`;
    94→    return fmtDate(d);
    95→  } catch {
    96→    return d;
    97→  }

   315→/* ── SVG Sparkline Component ── */
   316→function Sparkline({ data, color = '#10b981', width = 80, height = 28 }: { data: number[]; color?: string; width?: number; height?: number }) {
   317→  if (data.length < 2) return null;
   318→  const min = Math.min(...data);
   319→  const max = Math.max(...data);
   320→  const range = max - min || 1;
   321→  const padding = 2;
   322→  const w = width - padding * 2;
   323→  const h = height - padding * 2;
   324→
   325→  const points = data.map((v, i) => {
   326→    const x = padding + (i / (data.length - 1)) * w;
   327→    const y = padding + h - ((v - min) / range) * h;
   328→    return `${x},${y}`;
   329→  });
   330→
   331→  const areaPoints = [...points, `${padding + w},${padding + h}`, `${padding},${padding + h}`];
   332→
   333→  const gradientId = `spark-${color.replace('#', '')}`;
   334→
   335→  return (
   336→    <svg width={width} height={height} className="overflow-visible">
   337→      <defs>
   338→        <linearGradient id={gradientId} x1="0%" y1="0%" x2="0%" y2="100%">
   339→          <stop offset="0%" stopColor={color} stopOpacity="0.25" />
   340→          <stop offset="100%" stopColor={color} stopOpacity="0.02" />
   341→        </linearGradient>
   342→      </defs>
   343→      <polygon points={areaPoints.join(' ')} fill={`url(#${gradientId})`} />
   344→      <polyline
   345→        points={points.join(' ')}
   346→        fill="none"
   347→        stroke={color}
   348→        strokeWidth="1.5"
   349→        strokeLinejoin="round"
   350→        strokeLinecap="round"
   351→      />
   352→      {/* Last point highlight */}
   353→      {points.length > 0 && (() => {
   354→        const lastPt = points[points.length - 1].split(',');
   355→        return <circle cx={lastPt[0]} cy={lastPt[1]} r="2.5" fill={color} stroke="white" strokeWidth="1.5" />;
   356→      })()}
   357→    </svg>
   358→  );
   359→}

   437→function buildActivityEvents(planillas: DashboardData['planillas_recientes']): ActivityEvent[] {
   438→  const events: ActivityEvent[] = [];
   439→
   440→  planillas.forEach(p => {
   441→    events.push({
   442→      id: `${p.id}-creation`,
   443→      type: 'creation',
   444→      description: `Planilla ${p.codigo} creada`,
   445→      timestamp: p.fecha_creacion,
   446→      status: 'BORRADOR',
   447→      codigo: p.codigo,
   448→    });
   449→
   450→    if (p.estado !== 'BORRADOR') {
   451→      events.push({
   452→        id: `${p.id}-calc`,
   453→        type: 'calculation',
   454→        description: `Planilla ${p.codigo} calculada — ${p.empleados} empleados, ${fmt(p.total_neto)}`,
   455→        timestamp: p.fecha_creacion,
   456→        status: 'CALCULADA',
   457→        codigo: p.codigo,
   458→      });
   459→    }
   460→
   461→    if (p.estado === 'APROBADA' || p.estado === 'PAGADA') {
   462→      events.push({
   463→        id: `${p.id}-approval`,
   464→        type: 'approval',
   465→        description: `Planilla ${p.codigo} aprobada${p.aprobada_por ? ` por ${p.aprobada_por}` : ''}`,
   466→        timestamp: p.fecha_creacion,
   467→        status: 'APROBADA',
   468→        codigo: p.codigo,
   469→      });
   470→    }
   471→
