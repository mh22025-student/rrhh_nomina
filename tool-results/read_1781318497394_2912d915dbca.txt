    92→/* ─── circular progress ring ─── */
    93→function CircularProgress({ value, max, size = 80, strokeWidth = 6, color = 'emerald' }: {
    94→  value: number; max: number; size?: number; strokeWidth?: number; color?: string;
    95→}) {
    96→  const radius = (size - strokeWidth) / 2;
    97→  const circumference = radius * 2 * Math.PI;
    98→  const pct = max > 0 ? Math.min(value / max, 1) : 0;
    99→  const offset = circumference - pct * circumference;
   100→  const colorMap: Record<string, string> = {
   101→    emerald: 'text-emerald-500',
   102→    amber: 'text-amber-500',
   103→    sky: 'text-sky-500',
   104→    purple: 'text-purple-500',
   105→  };
   106→  const strokeClass = colorMap[color] || colorMap.emerald;
   107→  return (
   108→    <svg width={size} height={size} className="-rotate-90">
   109→      <circle cx={size / 2} cy={size / 2} r={radius} fill="none" stroke="currentColor" strokeWidth={strokeWidth} className="text-slate-200 dark:text-slate-700" />
   110→      <circle cx={size / 2} cy={size / 2} r={radius} fill="none" stroke="currentColor" strokeWidth={strokeWidth} strokeDasharray={circumference} strokeDashoffset={offset} strokeLinecap="round" className={`${strokeClass} transition-all duration-700`} />
   111→    </svg>
   112→  );
   113→}
