    89→function calcDaysBetween(start: string, end: string): number {
    90→  if (!start || !end) return 0;
    91→  const s = new Date(start);
    92→  const e = new Date(end);
    93→  const diff = Math.ceil((e.getTime() - s.getTime()) / (1000 * 60 * 60 * 24)) + 1;
    94→  return diff > 0 ? diff : 0;
    95→}

    97→function calcTenure(fechaIngreso: string): { years: number; months: number } {
    98→  const start = new Date(fechaIngreso);
    99→  const now = new Date();
   100→  let years = now.getFullYear() - start.getFullYear();
   101→  let months = now.getMonth() - start.getMonth();
   102→  if (months < 0) {
   103→    years--;
   104→    months += 12;
   105→  }
   106→  if (now.getDate() < start.getDate()) {
   107→    months--;
   108→    if (months < 0) {
   109→      years--;
   110→      months += 12;
   111→    }
   112→  }
   113→  return { years: Math.max(0, years), months: Math.max(0, months) };
   114→}

   116→// Circular Progress Component
   117→function CircularProgress({ value, size = 80, strokeWidth = 8, colorClass = 'text-emerald-500', trackClass = 'text-slate-100 dark:text-slate-700', children }: {
   118→  value: number; size?: number; strokeWidth?: number;
   119→  colorClass?: string; trackClass?: string;
   120→  children?: React.ReactNode;
   121→}) {
   122→  const radius = (size - strokeWidth) / 2;
   123→  const circumference = 2 * Math.PI * radius;
   124→  const offset = circumference * (1 - value / 100);
   125→  return (
   126→    <div className="relative shrink-0" style={{ width: size, height: size }}>
   127→      <svg className="-rotate-90" style={{ width: size, height: size }} viewBox={`0 0 ${size} ${size}`}>
   128→        <circle cx={size / 2} cy={size / 2} r={radius} fill="none" stroke="currentColor" strokeWidth={strokeWidth} className={trackClass} />
   129→        <circle
   130→          cx={size / 2} cy={size / 2} r={radius} fill="none"
   131→          stroke="currentColor" strokeWidth={strokeWidth}
   132→          strokeDasharray={circumference}
   133→          strokeDashoffset={offset}
   134→          strokeLinecap="round"
   135→          className={`${colorClass} transition-all duration-700`}
   136→        />
   137→      </svg>
   138→      {children && (
   139→        <div className="absolute inset-0 flex flex-col items-center justify-center">
   140→          {children}
   141→        </div>
   142→      )}
   143→    </div>
   144→  );
   145→}

   147→// Mini bar chart for salary trend
   148→function SalaryBarChart({ data: chartData }: { data: { label: string; value: number }[] }) {
   149→  if (chartData.length === 0) return null;
   150→  const maxVal = Math.max(...chartData.map(d => d.value));
   151→  return (
   152→    <div className="flex items-end gap-1.5 h-16">
   153→      {chartData.map((d, i) => {
   154→        const height = maxVal > 0 ? (d.value / maxVal) * 100 : 0;
   155→        return (
   156→          <div key={i} className="flex-1 flex flex-col items-center gap-0.5">
   157→            <div
   158→              className="w-full rounded-t-sm bg-gradient-to-t from-emerald-500 to-teal-400 dark:from-emerald-600 dark:to-teal-500 transition-all duration-500 min-h-[4px]"
   159→              style={{ height: `${Math.max(height, 8)}%` }}
   160→              title={`${d.label}: ${fmt(d.value)}`}
   161→            />
   162→            <span className="text-[8px] text-slate-400 dark:text-slate-500 truncate w-full text-center">{d.label}</span>
   163→          </div>
   164→        );
   165→      })}
   166→    </div>
   167→  );
   168→}
