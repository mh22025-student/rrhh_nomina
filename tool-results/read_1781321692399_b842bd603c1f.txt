   304→  const [lockoutCountdown, setLockoutCountdown] = useState('');
   305→
   306→  // Lockout timer effect
   307→  useEffect(() => {
   308→    if (!lockoutUntil) return;
   309→    const interval = setInterval(() => {
   310→      const remaining = lockoutUntil - Date.now();
   311→      if (remaining <= 0) {
   312→        setLockoutUntil(null);
   313→        setFailedAttempts(0);
   314→        setLockoutCountdown('');
   315→        clearInterval(interval);
   316→        return;
   317→      }
   318→      const mins = Math.floor(remaining / 60000);
   319→      const secs = Math.floor((remaining % 60000) / 1000);
   320→      setLockoutCountdown(`${mins}:${secs.toString().padStart(2, '0')}`);

   331→      await onLogin(email, password);

   334→      setFailedAttempts(newAttempts);
   335→      const errorMsg = err instanceof Error ? err.message : 'Error desconocido';

   337→      // Lockout after 5 failed attempts

   339→        setLockoutUntil(Date.now() + 5 * 60 * 1000); // 5 minute lockout

   655→// ============================================================
   656→// PASSWORD RECOVERY DIALOG
   657→// ============================================================
   658→interface PasswordRecoveryDialogProps {
   659→  open: boolean;
   660→  onOpenChange: (open: boolean) => void;
   661→}
   662→
   663→function PasswordRecoveryDialog({ open, onOpenChange }: PasswordRecoveryDialogProps) {
   664→  const [step, setStep] = useState(1);
   665→  const [email, setEmail] = useState('');
   666→  const [otp, setOtp] = useState('');
   667→  const [otpId, setOtpId] = useState('');
   668→  const [newPassword, setNewPassword] = useState('');
   669→  const [confirmPassword, setConfirmPassword] = useState('');
   670→  const [demoOtp, setDemoOtp] = useState<string | null>(null);
   671→  const [isLoading, setIsLoading] = useState(false);
   672→  const [error, setError] = useState<string | null>(null);
   673→  const [success, setSuccess] = useState<string | null>(null);
   674→  const { toast } = useToast();
   675→
   676→  const handleReset = () => {
   677→    setStep(1);
   678→    setEmail('');
   679→    setOtp('');
   680→    setOtpId('');
   681→    setNewPassword('');
   682→    setConfirmPassword('');
   683→    setDemoOtp(null);
   684→    setIsLoading(false);
   685→    setError(null);
   686→    setSuccess(null);
   687→  };
   688→
   689→  const handleClose = (newOpen: boolean) => {
   690→    if (!newOpen) handleReset();
   691→    onOpenChange(newOpen);
   692→  };
   693→
   694→  const handleStep1 = async () => {
   695→    setError(null);
   696→    setIsLoading(true);
   697→    try {
   698→      const res = await fetch('/api/auth/forgot-password', {
   699→        method: 'POST',
   700→        headers: { 'Content-Type': 'application/json' },
   701→        body: JSON.stringify({ email }),
   702→      });
   703→      const data = await res.json();
   704→      if (!res.ok) {
   705→        setError(data.error || 'Error al solicitar código');
   706→        return;
   707→      }
   708→      setDemoOtp(data.demo_otp);
   709→      setSuccess('Código enviado. Verifique su correo.');
   710→      setStep(2);
   711→    } catch {
   712→      setError('Error de conexión');
   713→    } finally {
   714→      setIsLoading(false);
   715→    }
   716→  };
   717→
   718→  const handleStep2 = async () => {
   719→    setError(null);
   720→    setIsLoading(true);
   721→    try {
   722→      const res = await fetch('/api/auth/verify-otp', {
   723→        method: 'POST',
   724→        headers: { 'Content-Type': 'application/json' },
   725→        body: JSON.stringify({ email, otp }),
   726→      });
   727→      const data = await res.json();
   728→      if (!res.ok) {
   729→        setError(data.error || 'Código inválido');
   730→        return;
   731→      }
   732→      setOtpId(data.otp_id);
   733→      setSuccess('Código verificado');
   734→      setStep(3);
   735→    } catch {
   736→      setError('Error de conexión');
   737→    } finally {
   738→      setIsLoading(false);
   739→    }
   740→  };
   741→
   742→  const handleStep3 = async () => {
   743→    setError(null);
   744→    if (newPassword !== confirmPassword) {
   745→      setError('Las contraseñas no coinciden');
   746→      return;
   747→    }
   748→    if (newPassword.length < 8) {
   749→      setError('La contraseña debe tener al menos 8 caracteres');
   750→      return;
   751→    }
   752→    setIsLoading(true);
   753→    try {
   754→      const res = await fetch('/api/auth/reset-password', {
   755→        method: 'POST',
   756→        headers: { 'Content-Type': 'application/json' },
   757→        body: JSON.stringify({ otp_id: otpId, email, new_password: newPassword }),
   758→      });
   759→      const data = await res.json();
   760→      if (!res.ok) {
   761→        setError(data.error || 'Error al actualizar contraseña');
   762→        return;
   763→      }
   764→      toast({
   765→        title: 'Contraseña actualizada',
   766→        description: 'Puede iniciar sesión con su nueva contraseña',
   767→      });
   768→      handleClose(false);
   769→    } catch {
   770→      setError('Error de conexión');
   771→    } finally {
   772→      setIsLoading(false);
   773→    }
   774→  };
   775→
   776→  const stepLabels = ['Email', 'Verificación', 'Nueva Contraseña'];
   777→  const progressValue = (step / 3) * 100;
   778→
   779→  return (
   780→    <Dialog open={open} onOpenChange={handleClose}>
   781→      <DialogContent className="sm:max-w-md">
   782→        <DialogHeader>
   783→          <DialogTitle className="flex items-center gap-2">
   784→            <Lock className="h-5 w-5 text-emerald-600" />
   785→            Recuperar Contraseña
   786→          </DialogTitle>
   787→          <DialogDescription>
   788→            Siga los pasos para restablecer su contraseña
   789→          </DialogDescription>
   790→        </DialogHeader>
   791→
   792→        {/* Progress indicator */}
   793→        <div className="space-y-2">
   794→          <div className="flex justify-between text-xs text-slate-500">
   795→            {stepLabels.map((label, i) => (
   796→              <span key={i} className={step > i ? 'text-emerald-600 font-medium' : step === i + 1 ? 'text-slate-900 font-medium' : ''}>
   797→                {i + 1}. {label}
   798→              </span>
   799→            ))}
   800→          </div>
   801→          <Progress value={progressValue} className="h-2" />
   802→        </div>
   803→
   804→        <div className="space-y-4">
   805→          {error && (
   806→            <div className="flex items-center gap-2 text-sm text-red-600 bg-red-50 p-3 rounded-lg">
   807→              <AlertCircle className="h-4 w-4 shrink-0" />
   808→              <span>{error}</span>
   809→            </div>
   810→          )}
   811→
   812→          {success && step > 1 && (
   813→            <div className="flex items-center gap-2 text-sm text-emerald-700 bg-emerald-50 p-3 rounded-lg">
   814→              <CheckCircle className="h-4 w-4 shrink-0" />
   815→              <span>{success}</span>
   816→            </div>
   817→          )}
   818→
   819→          {/* Step 1: Email */}
   820→          {step === 1 && (
   821→            <div className="space-y-4">
   822→              <div className="space-y-2">
   823→                <Label>Correo Electrónico</Label>
   824→                <Input
   825→                  type="email"
   826→                  placeholder="usuario@nomina.gob.sv"
   827→                  value={email}
   828→                  onChange={(e) => setEmail(e.target.value)}
   829→                  disabled={isLoading}
   830→                />
   831→              </div>
   832→              <Button onClick={handleStep1} disabled={isLoading || !email} className="w-full bg-emerald-600 hover:bg-emerald-700">
   833→                {isLoading ? <Loader2 className="mr-2 h-4 w-4 animate-spin" /> : null}
   834→                Enviar Código
   835→              </Button>
   836→            </div>
   837→          )}
   838→
   839→          {/* Step 2: OTP */}
   840→          {step === 2 && (
   841→            <div className="space-y-4">
   842→              <div className="space-y-2">
   843→                <Label>Código de Verificación</Label>
   844→                <Input
   845→                  placeholder="000000"
   846→                  value={otp}
   847→                  onChange={(e) => setOtp(e.target.value.replace(/\D/g, '').slice(0, 6))}
   848→                  disabled={isLoading}
   849→                  className="text-center text-lg tracking-[0.5em] font-mono"
   850→                  maxLength={6}
   851→                />
   852→              </div>
   853→              {demoOtp && (
   854→                <p className="text-xs text-amber-700 bg-amber-50 p-2 rounded">
   855→                  Demo - Código: <strong>{demoOtp}</strong>
   856→                </p>
   857→              )}
   858→              <div className="flex gap-2">
   859→                <Button variant="outline" onClick={() => setStep(1)} className="flex-1">
   860→                  <ArrowLeft className="mr-1 h-4 w-4" /> Volver
   861→                </Button>
   862→                <Button onClick={handleStep2} disabled={isLoading || otp.length !== 6} className="flex-1 bg-emerald-600 hover:bg-emerald-700">
   863→                  {isLoading ? <Loader2 className="mr-2 h-4 w-4 animate-spin" /> : null}
   864→                  Verificar
   865→                </Button>
   866→              </div>
   867→            </div>
   868→          )}
   869→
   870→          {/* Step 3: New Password */}
   871→          {step === 3 && (
   872→            <div className="space-y-4">
   873→              <div className="space-y-2">
   874→                <Label>Nueva Contraseña</Label>
   875→                <Input
   876→                  type="password"
   877→                  placeholder="Mínimo 8 caracteres"
   878→                  value={newPassword}
   879→                  onChange={(e) => setNewPassword(e.target.value)}
   880→                  disabled={isLoading}
   881→                />
   882→              </div>
   883→              <div className="space-y-2">
   884→                <Label>Confirmar Contraseña</Label>
   885→                <Input
   886→                  type="password"
   887→                  placeholder="Repita la contraseña"
